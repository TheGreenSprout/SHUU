WARNING

When making devconsole commands, follow this structure:

[DevConsoleCommand("command", "description")]
public static (string[], Color?) ClearSet(parameters)
{
    
}

string[] are the output lines that will be displayed on the console,
with Color? being the color of these lines (if left null the lines will be white)


also, if you want an array of elements as a parameter, currently only "params" at the end of the parameters is supported:

[DevConsoleCommand("command", "description")]
public static (string[], Color?) ClearSet(int num1, int num2, params int[] numarray)
{
    
}


theres also 2 custom parameter types: OptionalParameter<T> and MutableParameter

[DevConsoleCommand("command", "description")]
public static (string[], Color?) ClearSet(OptionalParameter<float> optional, MutableParameter mutable)
{
    // OPTIONAL
    if (!optional.TryGetValue(out float _optional)) // method handles _optional;


    // MUTABLE
    if (mutable.TryGetValue<bool>(out bool b)) ;
    else if (mutable.TryGetValue<int>(out int i)) ;
    else if (mutable.TryGetValue<float>(out float f)) ;
    else if (mutable.TryGetValue<string>(out string s)) ;
    else if (mutable.TryGetValue<char>(out char c)) returnVal = ;
    else returnVal = (new string[] { "Unsupported value type" }, Color.red);
}
